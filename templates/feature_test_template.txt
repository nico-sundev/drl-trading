    feature: ${FeatureName}Feature,

    expected_values = [50, 55, 60, 65, 70, 75]

    # Create a Series with the same index as prepared_source_df
    feature_series = Series(expected_values, index=prepared_source_df.index)
    patched_${feature_name}.return_value = feature_series

    # When
    result = feature.compute()

    # Then
    patched_${feature_name}.assert_called_once_with(
        prepared_source_df["Close"],
        # Uncomment and adjust parameters as needed for your specific feature
        # prepared_source_df["High"],
        # prepared_source_df["Low"],
        length=config.length,
    )

    assert isinstance(result.index, DatetimeIndex)
    assert "${feature_name}_14test" in result.columns
    assert result["${feature_name}_14test"].head(6).tolist() == expected_values
